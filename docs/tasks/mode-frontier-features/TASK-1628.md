### 3. deck.gl GridLayer 描画 🔵

**信頼性**: 🔵 *design-interview.md Q6・architecture.md deck.gl GridLayer選択より*

```typescript
const cacheKey = `${surrogateModelType}_${param1}_${param2}_${objective}_50`
const result = surface3dCache.get(cacheKey)

// GridLayer用データ変換: heatmap[i*nGrid+j] -> {position: [grid1[j], grid2[i]], value: heatmap[i*nGrid+j]}
const gridData = result
  ? result.heatmap.map((val, idx) => {
      const i = Math.floor(idx / result.nGrid)
      const j = idx % result.nGrid
      return { position: [result.grid1[j], result.grid2[i]], value: val }
    })
  : []

const layers = [
  new GridLayer({
    id: 'surface3d-layer',
    data: gridData,
    getPosition: d => d.position,
    getElevationWeight: d => d.value,
    elevationAggregation: 'MEAN',
    cellSize: 1,
    extruded: true,
    pickable: true,
  })
]
```
