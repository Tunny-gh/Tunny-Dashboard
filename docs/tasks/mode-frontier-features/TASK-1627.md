### 4. computeSurface3d アクション 🔵

**信頼性**: 🔵 *dataflow.md surface3dシーケンス図より*

```typescript
computeSurface3d: async (param1, param2, objective, nGrid = 50) => {
  const { surrogateModelType, surface3dCache } = get()
  const cacheKey = `${surrogateModelType}_${param1}_${param2}_${objective}_${nGrid}`

  // キャッシュヒット確認
  if (surface3dCache.has(cacheKey)) {
    return
  }

  set({ isComputingSurface: true, surface3dError: null })
  try {
    const raw = wasmLoader.computePdp2d(param1, param2, objective, nGrid)
    const result: Surface3DResult = {
      param1,
      param2,
      objective,
      grid1: Array.from(raw.grid1),
      grid2: Array.from(raw.grid2),
      heatmap: Array.from(raw.heatmap),
      modelType: surrogateModelType,
      nGrid,
      durationMs: raw.durationMs,
    }
    const newCache = new Map(surface3dCache)
    newCache.set(cacheKey, result)
    set({ surface3dCache: newCache, isComputingSurface: false })
  } catch (err) {
    console.error('[analysisStore] computeSurface3d error:', err)
    set({
      surface3dError: err instanceof Error ? err.message : 'Surface3D computation failed',
      isComputingSurface: false,
    })
  }
},
```
