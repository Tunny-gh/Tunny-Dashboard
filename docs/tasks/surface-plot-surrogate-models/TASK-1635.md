### 1. L-BFGS Two-loop recursion 🔵

**信頼性**: 🔵 *architecture.md「L-BFGS (m=5) Two-loop recursion」より*

**実装ファイル**: `rust_core/src/kriging.rs`

```rust
/// L-BFGS Two-loop recursion で探索方向 d = -H^{-1} grad を計算
/// s_hist[k] = x_{k+1} - x_k
/// y_hist[k] = grad_{k+1} - grad_k
pub(crate) fn lbfgs_direction(
    grad: &[f64],
    s_hist: &[Vec<f64>],
    y_hist: &[Vec<f64>],
) -> Vec<f64> {
    let m = s_hist.len();
    let mut q = grad.to_vec();
    let mut rho = vec![0.0; m];
    let mut alpha = vec![0.0; m];

    // First loop (backward)
    for i in (0..m).rev() {
        let sy: f64 = s_hist[i].iter().zip(&y_hist[i]).map(|(s, y)| s * y).sum();
        if sy.abs() < 1e-15 { continue; }
        rho[i] = 1.0 / sy;
        alpha[i] = rho[i] * s_hist[i].iter().zip(&q).map(|(s, qi)| s * qi).sum::<f64>();
        for (qi, yi) in q.iter_mut().zip(&y_hist[i]) {
            *qi -= alpha[i] * yi;
        }
    }

    // Initial Hessian approximation: H_0 = (s_m^T y_m) / (y_m^T y_m) * I
    let gamma = if m > 0 {
        let sy: f64 = s_hist[m-1].iter().zip(&y_hist[m-1]).map(|(s, y)| s * y).sum();
        let yy: f64 = y_hist[m-1].iter().map(|y| y * y).sum();
        if yy > 1e-15 { sy / yy } else { 1.0 }
    } else { 1.0 };
    let mut r: Vec<f64> = q.iter().map(|qi| gamma * qi).collect();

    // Second loop (forward)
    for i in 0..m {
        let yr: f64 = y_hist[i].iter().zip(&r).map(|(y, ri)| y * ri).sum();
        let beta = rho[i] * yr;
        for (ri, si) in r.iter_mut().zip(&s_hist[i]) {
            *ri += (alpha[i] - beta) * si;
        }
    }

    r.iter_mut().for_each(|v| *v = -*v);  // d = -H^{-1} grad
    r
}
```

### 2. Armijo バックトラッキング線探索 🔵

**信頼性**: 🔵 *architecture.md「Armijo バックトラッキング（c₁=1e-4）」より*

```rust
/// Armijo バックトラッキング線探索
/// 十分減少条件: f(x + α·d) ≤ f(x) + c₁·α·(grad^T d)
pub(crate) fn armijo_line_search(
    f_x: f64,
    grad: &[f64],
    d: &[f64],
    f: impl Fn(&[f64]) -> f64,
    x: &[f64],
    c1: f64,
    max_iter: usize,
) -> f64 {
    let slope: f64 = grad.iter().zip(d).map(|(g, di)| g * di).sum();
    let mut alpha = 1.0;
    for _ in 0..max_iter {
        let x_new: Vec<f64> = x.iter().zip(d).map(|(xi, di)| xi + alpha * di).collect();
        let f_new = f(&x_new);
        if f_new <= f_x + c1 * alpha * slope {
            return alpha;
        }
        alpha *= 0.5;
    }
    alpha
}
```

### 3. ハイパーパラメータ最適化 🔵

**信頼性**: 🔵 *architecture.md「最大 100 イテレーション、収束判定 ‖∇L‖ < 1e-5」より*

```rust
/// L-BFGS でハイパーパラメータを最適化
/// params = [log_ls_0, ..., log_ls_{d-1}, log_sf, log_sn]
pub(crate) fn optimize_hyperparams(
    x: &[Vec<f64>],
    y: &[f64],
    n_iter: usize,
    m_history: usize,  // L-BFGS 履歴数（デフォルト: 5）
) -> Vec<f64> {
    let ndim = if x.is_empty() { return vec![]; } else { x[0].len() };
    let mut params = vec![0.0; ndim + 2];  // log_ls×d, log_sf, log_sn
    params[ndim + 1] = -2.0;  // log_sn 初期値: -2.0 → σ_n ≈ 0.135

    let mut s_hist: Vec<Vec<f64>> = Vec::new();
    let mut y_hist: Vec<Vec<f64>> = Vec::new();

    let f = |p: &[f64]| {
        let lml = log_marginal_likelihood(x, y, &p[..ndim], p[ndim], p[ndim+1]);
        -lml  // 最小化するので符号反転
    };

    for _ in 0..n_iter {
        let grad_neg: Vec<f64> = log_ml_gradient(x, y, &params)
            .iter().map(|g| -g).collect();  // -∇L（最小化の勾配）

        // 収束判定
        let grad_norm: f64 = grad_neg.iter().map(|g| g * g).sum::<f64>().sqrt();
        if grad_norm < 1e-5 { break; }

        let d = lbfgs_direction(&grad_neg, &s_hist, &y_hist);
        let f_x = f(&params);
        let alpha = armijo_line_search(f_x, &grad_neg, &d, &f, &params, 1e-4, 20);

        let x_new: Vec<f64> = params.iter().zip(&d).map(|(p, di)| p + alpha * di).collect();
        let grad_new: Vec<f64> = log_ml_gradient(x, y, &x_new)
            .iter().map(|g| -g).collect();

        let s: Vec<f64> = x_new.iter().zip(&params).map(|(xn, xo)| xn - xo).collect();
        let y: Vec<f64> = grad_new.iter().zip(&grad_neg).map(|(gn, go)| gn - go).collect();

        params = x_new;

        if s_hist.len() >= m_history { s_hist.remove(0); y_hist.remove(0); }
        s_hist.push(s);
        y_hist.push(y);
    }
    params
}
```

### 4. GP 学習（train_gp）と予測（predict_mean） 🔵

**信頼性**: 🔵 *architecture.md「train_gp + n>1000 サブサンプリング + predict_mean」より*

```rust
pub(crate) fn train_gp(
    x: Vec<Vec<f64>>,
    y: Vec<f64>,
    subsample_n: usize,
    seed: u64,
) -> Option<GpModel> {
    let (x_sub, y_sub) = if x.len() > subsample_n {
        // ランダムサブサンプリング
        let mut rng = crate::rf::Lcg::new(seed);
        let n = x.len();
        let mut indices: Vec<usize> = (0..n).collect();
        for i in (1..n).rev() {
            let j = rng.next_usize(i + 1);
            indices.swap(i, j);
        }
        let idx = &indices[..subsample_n];
        let xs: Vec<Vec<f64>> = idx.iter().map(|&i| x[i].clone()).collect();
        let ys: Vec<f64> = idx.iter().map(|&i| y[i]).collect();
        (xs, ys)
    } else {
        (x, y)
    };

    let ndim = if x_sub.is_empty() { return None; } else { x_sub[0].len() };
    let params = optimize_hyperparams(&x_sub, &y_sub, 100, 5);
    let log_ls = params[..ndim].to_vec();
    let log_sf = params[ndim];
    let log_sn = params[ndim + 1];

    let k = build_kernel_matrix(&x_sub, &log_ls, log_sf, log_sn);
    let l = cholesky(&k)?;
    let alpha = compute_alpha(&l, &y_sub);

    Some(GpModel { alpha, x_train: x_sub, log_ls, log_sf, log_sn })
}

pub(crate) fn predict_mean(model: &GpModel, x_test: &[f64]) -> f64 {
    model.x_train.iter().zip(&model.alpha)
        .map(|(xi, ai)| ai * matern52_ard(x_test, xi, &model.log_ls, model.log_sf))
        .sum()
}
```
