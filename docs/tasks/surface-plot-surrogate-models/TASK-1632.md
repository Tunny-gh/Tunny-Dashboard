### 1. compute_pdp_2d_rf() 実装 🔵

**信頼性**: 🔵 *architecture.md 2D 部分空間射影アプローチ・dataflow.md RF PDP 計算フローより*

**実装ファイル**: `rust_core/src/pdp.rs`

```rust
use crate::rf;

pub(crate) fn compute_pdp_2d_rf(
    x_matrix: &[Vec<f64>],
    y: &[f64],
    param_names: &[String],
    objective_name: &str,
    param1_idx: usize,
    param2_idx: usize,
    n_grid: usize,
) -> PdpResult2d {
    let p1_name = param_names.get(param1_idx).cloned().unwrap_or_default();
    let p2_name = param_names.get(param2_idx).cloned().unwrap_or_default();
    let empty = PdpResult2d { /* ... */ };

    let n = y.len();
    if n < 2 || n_grid == 0 { return empty; }

    // 2D 部分空間射影: (param1, param2) 列のみ抽出
    let x_2d: Vec<Vec<f64>> = x_matrix.iter().map(|row| {
        vec![
            row.get(param1_idx).copied().unwrap_or(0.0),
            row.get(param2_idx).copied().unwrap_or(0.0),
        ]
    }).collect();

    // RF 学習（デフォルト: 100本、深さ10、min_samples_leaf=2、seed=42）
    let forest = rf::train_random_forest(&x_2d, y, 100, 10, 2, 42);

    // グリッド構築
    let col1: Vec<f64> = x_2d.iter().map(|r| r[0]).collect();
    let col2: Vec<f64> = x_2d.iter().map(|r| r[1]).collect();
    let min1 = col1.iter().cloned().fold(f64::INFINITY, f64::min);
    let max1 = col1.iter().cloned().fold(f64::NEG_INFINITY, f64::max);
    let min2 = col2.iter().cloned().fold(f64::INFINITY, f64::min);
    let max2 = col2.iter().cloned().fold(f64::NEG_INFINITY, f64::max);
    let grid1 = linspace(min1, max1, n_grid);
    let grid2 = linspace(min2, max2, n_grid);

    // グリッド予測
    let values: Vec<Vec<f64>> = grid1.iter().map(|&v1| {
        grid2.iter().map(|&v2| {
            rf::rf_predict(&forest, &[v1, v2])
        }).collect()
    }).collect();

    let r_squared = rf::rf_r_squared(&forest, &x_2d, y);

    PdpResult2d {
        param1_name: p1_name,
        param2_name: p2_name,
        objective_name: objective_name.to_string(),
        grid1,
        grid2,
        values,
        r_squared,
    }
}
```

pub fn compute_pdp_2d(
    param1_name: &str,
    param2_name: &str,
    objective_name: &str,
    n_grid: usize,
    model_type: &str,  // 追加
